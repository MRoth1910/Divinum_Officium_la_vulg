\begin{enpars}
\rubrique{Devotions follow according to the month or the occasion.}

January: Litany of the Holy Name, \pageref{lit_san_nom_en}.

March: Litany of Saint Joseph, \pageref{lit_san_jos_en}.

May and October: Litany of the Blessed Virgin Mary, \pageref{lit_bvm_en}.

June and on First Fridays: Litany of the Sacred Heart, \pageref{lit_sac_cor_en}.

July: Litany of the Precious Blood, \pageref{lit_pb_en}.
\end{enpars}
